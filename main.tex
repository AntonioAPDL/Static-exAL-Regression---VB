\section{Model, Augmentation, and Priors}

Fix \(p_0\in(0,1)\). For \(i=1,\dots,n\): \(y_i\in\mathbb{R}\), \(x_i\in\mathbb{R}^p\), \(\beta\in\mathbb{R}^p\), \(\sigma>0\), \(\gamma\in(L,U)\).
Let (dependence on \(p_0\) implicit)
\[
\begin{aligned}
g(\gamma) &= 2\Phi(-|\gamma|)e^{\gamma^2/2},\quad
p(\gamma)=\ind\{\gamma<0\}+\frac{p_0-\ind\{\gamma<0\}}{g(\gamma)},\\
A(\gamma) &= \frac{1-2p(\gamma)}{p(\gamma)[1-p(\gamma)]},\quad
B(\gamma) = \frac{2}{p(\gamma)[1-p(\gamma)]},\\
C(\gamma) &= \big(\ind\{\gamma>0\}-p(\gamma)\big)^{-1},\quad
\lambda(\gamma):=C(\gamma)\,|\gamma|\ge 0.
\end{aligned}
\]

Latents:
\[
s_i \stackrel{\text{iid}}{\sim} \TN(0,1),\qquad
v_i\mid\sigma \stackrel{\text{iid}}{\sim} \mathrm{Exp}\!\left(\text{rate}=1/\sigma\right).
\]

Observation model (independent over \(i\)):
\[
y_i \mid \beta,\sigma,\gamma,s_i,v_i \sim
\Normal\!\Big(x_i^\top\beta+\lambda(\gamma)\,\sigma\,s_i+A(\gamma)\,v_i,\;\; B(\gamma)\,\sigma\,v_i\Big).
\]

Priors:
\[
\beta\sim\Normal(b_0,V_0),\qquad
\sigma\sim\mathrm{IG}(a_\sigma,b_\sigma)\ \big[p(\sigma)\propto\sigma^{-(a_\sigma+1)}e^{-b_\sigma/\sigma}\big],\qquad
\gamma\sim\pi(\gamma)\ \text{on }(L,U).
\]
Let \(X=[x_1^\top;\dots;x_n^\top]\in\mathbb{R}^{n\times p}\), \(v=(v_1,\dots,v_n)^\top\), \(s=(s_1,\dots,s_n)^\top\), \(D(v)=\mathrm{diag}(v_1,\dots,v_n)\).

\section{Mean-Field Family and CAVI Objective}

We approximate
\[
q(\beta)\ \prod_{i=1}^n q(v_i)\ q(s_i)\ q(\sigma,\gamma),
\]
with \(q(\sigma,\gamma)\) \emph{joint}. Each factor satisfies
\[
\log q^\*(\cdot)\ \propto\ \E_{q(\text{others})}\!\left[\log \pi(\beta,\sigma,\gamma,s,v\mid y)\right].
\]

\paragraph{Scalar expectations from \(q(\sigma,\gamma)\).}
Define
\[
\boxed{
\begin{aligned}
\xi_1 &:= \E_{q(\sigma,\gamma)}\!\left[\frac{1}{B(\gamma)\,\sigma}\right],\qquad
\xi_{\lambda} := \E_{q(\sigma,\gamma)}\!\left[\frac{\lambda(\gamma)}{B(\gamma)}\right],\\
\xi_{\lambda^2} &:= \E_{q(\sigma,\gamma)}\!\left[\frac{\lambda(\gamma)^2\,\sigma}{B(\gamma)}\right],\qquad
\xi_A := \E_{q(\sigma,\gamma)}\!\left[\frac{A(\gamma)}{B(\gamma)\,\sigma}\right],\\
\xi_{A^2} &:= \E_{q(\sigma,\gamma)}\!\left[\frac{A(\gamma)^2}{B(\gamma)\,\sigma}\right],\qquad
\xi_{\sigma^{-1}} := \E_{q(\sigma,\gamma)}\!\left[\frac{1}{\sigma}\right],\\
\zeta_\lambda &:= \E_{q(\sigma,\gamma)}\!\left[\frac{\lambda(\gamma)\,A(\gamma)}{B(\gamma)}\right].
\end{aligned}}
\]
We also use, per \(i\): \(m_{v_i}^{(-1)}=\E[v_i^{-1}],\ m_{v_i}^{(1)}=\E[v_i],\ \mu_{s_i}=\E[s_i],\ m_{s_i}^{(2)}=\E[s_i^2]\).

\section{Closed-Form Variational Factors}

\subsection*{1) \(q(\beta)\) (Gaussian)}
\[
\boxed{
q(\beta)=\Normal(m_\beta,V_\beta),\quad
V_\beta=\Big(V_0^{-1}+\xi_1\,X^\top \mathrm{diag}(m_{v_1}^{(-1)},\ldots,m_{v_n}^{(-1)})\,X\Big)^{-1},
}
\]
\[
\boxed{
m_\beta=V_\beta\Big(
V_0^{-1}b_0
+X^\top\big[\ \xi_1\,\mathrm{diag}(m_{v}^{(-1)})\,y
-\xi_{\lambda}\,(m_{v}^{(-1)}\odot \mu_s)
-\xi_{A}\,\mathbf{1}_n\ \big]\Big),
}
\]
where \(\mu_s=(\mu_{s_1},\ldots,\mu_{s_n})^\top\) and \(\odot\) is Hadamard product.

\subsection*{2) \(q(v_i)\) (Generalized Inverse Gaussian; independent over \(i\))}
We use \(\GIG(k,\chi,\psi)\) with density \(f(z)\propto z^{k-1}\exp\{-(\chi/z+\psi z)/2\}\).
\[
\boxed{
q(v_i)=\GIG\!\big(\tfrac12,\ \chi_i,\ \psi\big),\qquad
\psi=\xi_{A^2}+2\,\xi_{\sigma^{-1}},
}
\]
\[
\boxed{
\chi_i=\xi_1\!\left(y_i^2-2y_i\,x_i^\top m_\beta+(x_i^\top m_\beta)^2+x_i^\top V_\beta x_i\right)
-2\,\xi_\lambda\,y_i\,\mu_{s_i}
+\xi_{\lambda^2}\,m_{s_i}^{(2)}
+2\,\xi_\lambda\,(x_i^\top m_\beta)\,\mu_{s_i}.
}
\]

\emph{GIG moments.} For \(v\sim\GIG(k,\chi,\psi)\),
\[
\E[v^r]=\left(\sqrt{\frac{\chi}{\psi}}\right)^{\!r}
\frac{K_{k+r}(\sqrt{\chi\psi})}{K_{k}(\sqrt{\chi\psi})},\quad
m_{v_i}^{(-1)}=\E[v_i^{-1}],\ \ m_{v_i}^{(1)}=\E[v_i].
\]

\subsection*{3) \(q(s_i)\) (Truncated Normal on \((0,\infty)\); independent over \(i\))}
\[
\boxed{
q(s_i)=\Normal(\mu_{s_i},\tau_{s_i}^2)\ \text{truncated to }(0,\infty),
}
\]
\[
\boxed{
\tau_{s_i}^2=\Big(1+\xi_{\lambda^2}\,m_{v_i}^{(-1)}\Big)^{-1},\qquad
\mu_{s_i}= \tau_{s_i}^2\Big(\ \xi_{\lambda}\,m_{v_i}^{(-1)}(y_i-x_i^\top m_\beta)\ -\ \zeta_\lambda\ \Big).
}
\]

\emph{TN moments.} With \(\alpha=\mu_{s_i}/\tau_{s_i}\) and \(\Lambda=\phi(\alpha)/\Phi(\alpha)\):
\[
\E[s_i]=\mu_{s_i}+\tau_{s_i}\Lambda,\qquad
\E[s_i^2]=\tau_{s_i}^2+\mu_{s_i}^2+\tau_{s_i}\mu_{s_i}\Lambda.
\]

\subsection*{4) \(q(\sigma,\gamma)\) (joint, kernel up to proportionality)}
\[
\boxed{
\begin{aligned}
\log q(\sigma,\gamma)\ \propto\ 
&\ \log \pi(\gamma)\ -\ \frac{n}{2}\log B(\gamma)\ -\ \Big(\tfrac{3n}{2}+a_\sigma+1\Big)\log\sigma\\
&\ -\ \frac{1}{2\,B(\gamma)\,\sigma}\sum_{i=1}^n
\E_{q(\beta),q(v_i)}\!\left[\frac{(y_i-x_i^\top\beta-A(\gamma)v_i)^2}{v_i}\right]
-\frac{1}{\sigma}\sum_{i=1}^n m_{v_i}^{(1)}-\frac{b_\sigma}{\sigma}\\
&\ +\ \frac{\lambda(\gamma)}{B(\gamma)}\sum_{i=1}^n
\E_{q(\beta),q(s_i),q(v_i)}\!\left[\frac{s_i(y_i-x_i^\top\beta-A(\gamma)v_i)}{v_i}\right]\\
&\ -\ \frac{\lambda(\gamma)^2}{2\,B(\gamma)}\,\sigma \sum_{i=1}^n
\E_{q(s_i),q(v_i)}\!\left[\frac{s_i^2}{v_i}\right].
\end{aligned}}
\]
The expectations inside are explicit given current moments:
\[
\begin{aligned}
\E_{q(\beta),q(v_i)}\!\left[\frac{(y_i-x_i^\top\beta-A(\gamma)v_i)^2}{v_i}\right]
&= (y_i-x_i^\top m_\beta)^2 + x_i^\top V_\beta x_i -2(y_i-x_i^\top m_\beta)A(\gamma)+A(\gamma)^2,\\
\E_{q(\beta),q(s_i),q(v_i)}\!\left[\frac{s_i(y_i-x_i^\top\beta-A(\gamma)v_i)}{v_i}\right]
&= \mu_{s_i}\,m_{v_i}^{(-1)}(y_i-x_i^\top m_\beta) - \mu_{s_i}A(\gamma),\\
\E_{q(s_i),q(v_i)}\!\left[\frac{s_i^2}{v_i}\right]
&= m_{s_i}^{(2)}\,m_{v_i}^{(-1)}.
\end{aligned}
\]

\section{CAVI Iteration (One Pass)}
\begin{enumerate}
\item Update \(q(\beta)=\Normal(m_\beta,V_\beta)\) using the boxed formulas in \S3.1.
\item For each \(i\): update \(q(v_i)=\GIG(\tfrac12,\chi_i,\psi)\), then compute \(m_{v_i}^{(-1)}\) and \(m_{v_i}^{(1)}\).
\item For each \(i\): update \(q(s_i)=\TN(\mu_{s_i},\tau_{s_i}^2)\), then compute \(\mu_{s_i}\) and \(m_{s_i}^{(2)}\).
\item Update joint \(q(\sigma,\gamma)\) via its kernel (e.g., Laplace/MH on \(\gamma\) + conditional update in \(\sigma\)); recompute \(\xi_1,\xi_\lambda,\xi_{\lambda^2},\xi_A,\xi_{A^2},\xi_{\sigma^{-1}},\zeta_\lambda\).
\item Repeat until ELBO convergence.
\end{enumerate}

\section{Support and Notation}
\[
\beta\in\mathbb{R}^p,\quad \sigma>0,\quad \gamma\in(L,U),\quad v_i>0,\ s_i>0.
\]
All expressions keep explicit dependence on \(A(\gamma),B(\gamma),C(\gamma),\lambda(\gamma)=C(\gamma)|\gamma|\).
